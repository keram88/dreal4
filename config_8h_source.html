125  // --------------------------------------------------------------------------
126  // NLopt options (stopping criteria)
127  // --------------------------------------------------------------------------
128  // These options are used when we use NLopt in refining
129  // counterexamples via local-optimization. The following
130  // descriptions are from
131  // https://nlopt.readthedocs.io/en/latest/NLopt_Reference/#stopping-criteria
132  //
133  // Set relative tolerance on function value: stop when an
134  // optimization step (or an estimate of the optimum) changes the
135  // objective function value by less than tol multiplied by the
136  // absolute value of the function value. (If there is any chance
137  // that your optimum function value is close to zero, you might want
138  // to set an absolute tolerance with nlopt_set_ftol_abs as well.)
139  // Criterion is disabled if tol is non-positive.
140  OptionValue<double> nlopt_ftol_rel_{1e-6};
141 
142  // Set absolute tolerance on function value: stop when an
143  // optimization step (or an estimate of the optimum) changes the
144  // function value by less than tol. Criterion is disabled if tol is
145  // non-positive.
146  OptionValue<double> nlopt_ftol_abs_{1e-6};
147 
148  // Stop when the number of function evaluations exceeds
149  // maxeval. (This is not a strict maximum: the number of function
150  // evaluations may exceed maxeval slightly, depending upon the
151  // algorithm.) Criterion is disabled if maxeval is non-positive.
152  OptionValue<int> nlopt_maxeval_{100};
153 
154  // Stop when the optimization time (in seconds) exceeds
155  // maxtime. (This is not a strict maximum: the time may exceed
156  // maxtime slightly, depending upon the algorithm and on how slow
157  // your function evaluation is.) Criterion is disabled if maxtime is
158  // non-positive.
159  OptionValue<double> nlopt_maxtime_{0.01};
160 
161  // Default initial phase (for PICOSAT):
